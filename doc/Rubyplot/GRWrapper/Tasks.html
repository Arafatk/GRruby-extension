def inqcolorfromrgb(rgbstring)
  rgb = rgbstring.match /#(..)(..)(..)/
  GR.inqcolorfromrgb((rgb[1].hex.to_f/255.0), (rgb[2].hex.to_f/255.0), (rgb[3].hex.to_f/255.0))
end
